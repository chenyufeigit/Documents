SELECT CONCAT('ALTER TABLE ', table_name,' CHARACTER SET UTF8 COLLATE utf8_bin', ';') FROM information_schema.COLUMNS WHERE TABLE_SCHEMA = '库名';
此语句会生成指定库中所有表的修改语句
如：
ALTER TABLE biao1 CHARACTER SET UTF8 COLLATE utf8_bin;
ALTER TABLE biao2 CHARACTER SET UTF8 COLLATE utf8_bin;
ALTER TABLE biaon CHARACTER SET UTF8 COLLATE utf8_bin;
